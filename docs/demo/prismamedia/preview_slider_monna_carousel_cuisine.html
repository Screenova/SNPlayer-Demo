PREVIEW format Monna in Slider with carousel on PrismaMedia

Format is show at the bottom left of the page.
It can be collapsed by clicking on the minus sign (-).

Add the monna slider in your page using this code in a <script> tag. 
				
var s = document.createElement("script");
s.src = "https://static.screenova.cloud/snplayer.js";
s.onload = function () {
	const SN = window.Screenova;

	const ScreenovaConfig = {
		placement: '6a3a43656ad3c657d3f8cc33',
		partnerId: '644250',
		width: 200,
		height: 340,
		format: 'slider',
		align:'left',
		hideButtons: 3000,
		autoroll: 6000,
		collapseButton: true,
		widget: "monna",
		mobile: {
			width: 360,
			height: 290,
		}
	}

	SN.init(ScreenovaConfig, (ad) => {
		if (ad && ad.subscribe){
			ad.subscribe(() => SN.tools.track("%%CLICK_URL_UNESC%%"), SN.VPAID_EVENTS.AdClickThru);
		  ad.subscribe(() => {
				if(window.top.coreAds){
					window.top.coreAds.refreshDisplaySlot("%%PATTERN:slotCode%%", { origine: "Screenova"});
					window.top.document.getElementById("core-ads-footer-web").style.setProperty("display", "block");
				}
			}, SN.VPAID_EVENTS.AdError);
		}
	});
}
document.body.appendChild(s);
