Monna in Slider with carousel and collapse action

Format is show at the bottom right of the page.
It can be collapsed by clicking on the minus sign (-).

Add the monna slider in your page using this code in a <script> tag. 
				
var s = document.createElement("script");
s.src = "https://static.screenova.cloud/snplayer.js";
s.onload = function () {
	const SN = window.Screenova;

	const ScreenovaConfig = {
		placement: '6a437c446ad3c657d3f8cc45',
		width: 200,
		height: 340,
		format: 'slider',
		hideButtons: 3000,
		autoroll: 6000,
		collapseButton: true,
		widget: "monna",
		mobile: {
			width: 360,
			height: 290,
		}
	}

	SN.init(ScreenovaConfig);
}
document.body.appendChild(s);
