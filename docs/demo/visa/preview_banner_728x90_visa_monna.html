<script type="text/javascript">
  (function () {
		const targetElement = document.currentScript.parentNode;
		const s = document.createElement("script");
		s.src = "https://static.screenova.cloud/snplayer.js";
		s.onload = function () {
			const SN = window.Screenova;

			const ScreenovaConfig = {
				placement: "6a3a43656ad3c657d3f8cc33",
				target: targetElement,
				width: '100%',
				height: 250,
				widget: "monna",
				autoroll: 6000,
				itemsMinSize: 200,
			};

			SN.init(ScreenovaConfig, (ad) => {
				if (ad && ad.subscribe){
					ad.subscribe(() => SN.tools.track("%%CLICK_URL_UNESC%%"), SN.VPAID_EVENTS.AdClickThru);
					ad.subscribe(() => {
						if(window.top.coreAds){
							window.top.coreAds.refreshDisplaySlot("%%PATTERN:slotCode%%", { origine: "Screenova"});
						}
					}, SN.VPAID_EVENTS.AdError);
				}
			});
		};
		document.body.appendChild(s);
	})();
</script>
