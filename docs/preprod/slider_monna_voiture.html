PREVIEW format Monna in Slider with carousel on PrismaMedia

Format is show at the bottom left of the page.
It can be collapsed by clicking on the minus sign (-).

Add the monna slider in your page using this code in a <script> tag. 
				

		var s = document.createElement("script");
		s.src = "https://static-preprod.screenova.cloud/snplayer.js";
		s.onload = function () {
			const SN = window.Screenova;

			const ScreenovaConfig = {
				placement: "6a313c955244dec81823bde9",
				partnerId: '644250',
				width: 200,
				height: 340,
				format: 'slider',
				align: 'left',
				hideButtons: 3000,
				autoroll: 6000,
				collapseButton: true,
				vmargin: 60,
				widget: "monna",
				mobile: {
					width: 360,
					height: 290,
				}
			}

			SN.init(ScreenovaConfig);
		}
		document.body.appendChild(s);
